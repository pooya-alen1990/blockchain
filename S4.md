#### Tickets Sol File
~~~~
pragma solidity ^0.5.6;

contract buyTicket {
    
    address payable owner;
    uint public tickets;
    uint price = 1 ether;
    
    mapping (address => uint) public kharidar;
    
    constructor() public{
        owner = msg.sender;
        tickets = 5;
    }
    
    function() payable external {
        kharid(1);
    }
    
    function kharid(uint a) public payable{
        
        if(msg.value != (price * a) || (a > tickets)){
            revert();
        }
        
        kharidar[msg.sender] += a;
        tickets -= a;
        
        if(tickets == 0){
            selfdestruct(owner);
        }
        
    }
}
~~~~
